171 void Knapsack<T>::solve()
172 {
173  size_t nb_obj = _Wt.size(); // number of objects
174  std::vector< std::vector<T> > K(nb_obj+1, std::vector<T>(_W+1,0));
175 
176  // Build table K[][] in bottom up mainner
177  for (size_t i = 0; i <= nb_obj; i++)
178  {
179  for (unsigned int w = 0; w <= _W; w++)
180  {
181  if (i==0 || w==0)
182  K[i][w] = 0;
183  else if (_Wt[i-1] <= w)
184  K[i][w] = std::max(_Val[i-1]+K[i-1][w-_Wt[i-1]], K[i-1][w]);
185  else
186  K[i][w] = K[i-1][w];
187  }
188  }
189  _opt_value = K[nb_obj][_W];
190 
191  // Create the object list
192  size_t i = nb_obj; unsigned int w = _W;
193  while (w > 0 && K[i][w]==K[i][w-1])
194  w--;
195  while (w > 0)
196  {
197  while (i > 0 && K[i][w]==K[i-1][w]) {
198  _Solution[i-1] = false;
199  i--;
200  }
201  w -= _Wt[i-1];
202  _Solution[i-1] = true;

204  }
