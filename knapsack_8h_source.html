207 template< class T >
208 inline void Knapsack<T>::solve()
209 {
210  size_t nb_obj = _Wt.size(); // number of objects
211  std::vector< std::vector<T> > K(nb_obj+1, std::vector<T>(_W+1,0));
212 
213  // Build table K[][] in bottom up mainner
214  for (size_t i = 0; i <= nb_obj; i++)
215  {
216  for (unsigned int w = 0; w <= _W; w++)
217  {
218  if (i==0 || w==0)
219  K[i][w] = 0;
220  else if (_Wt[i-1] <= w)
221  K[i][w] = std::max(_Val[i-1]+K[i-1][w-_Wt[i-1]], K[i-1][w]);
222  else
223  K[i][w] = K[i-1][w];
224  }
225  }
226  _opt_value = K[nb_obj][_W];
227 
228  // Create the object list
229  size_t i = nb_obj; unsigned int w = _W;
230  while (w > 0 && K[i][w]==K[i][w-1])
231  w--;
232  while (w > 0)
233  {
234  while (i > 0 && K[i][w]==K[i-1][w]) {
235  _Solution[i-1] = false;
236  i--;
237  }
238  w -= _Wt[i-1];
239  _Solution[i-1] = true;
240  i--;
241  }
242 }
